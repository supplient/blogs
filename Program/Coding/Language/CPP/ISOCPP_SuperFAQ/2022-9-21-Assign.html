Self-assignment problem:

class A {};

class T
{
private:
    A* p;
public:
    T() 
    {
        p = new A();
    }

    T* operator=(const T& t) 
    {
        delete p;
        p = new A{*(t.p)}; // !!! t.p is deleted if this == &t !!!
        return this;
    }
};

void assign(T& a, T& b) {
    a = b;
}
 
int main()
{
    T t;
    assign(t, t);
}

这一问题极少出现，但一旦出现就极难调试，因此应尽量避免写出这类代码。 性能要求不高的话，直接：

    T* operator=(const T& t) 
    {
        // Handle self-assign
        if(this == &t)
            return this;
        delete p;
        p = new A{*(t.p)}; // !!! t.p is deleted if this == &t !!!
        return this;
    }
